MariaDB [akademik]> create table mahasiswa (  //membuat tabel mahasiswa
    -> nim varchar(5) not null,  //deklarasi variabel nim bertipe data varchar
    -> nama varchar(50) not null, //deklarasi variabel nama bertipe data varchar
    -> jkel varchar(1) not null, //deklarasi variabel jenis kelamin bertipe data varchar
    -> alamat text, //deklarasi variabel alamat bertipe data teks (agar dapat memuat banyak)
    -> tgllhr date, //deklarasi variabel date bertipe data date (tgl - bulan - tahun)
    -> primary key(nim) //nim dijadikan sebagai primary key
    -> );
